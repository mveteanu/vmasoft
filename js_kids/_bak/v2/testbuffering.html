function collectFunctions()
{
    var o = {};

    for (var p in p5.prototype) 
    {
        if (typeof p5.prototype[p] === 'function') 
        {
            o[p] = window[p]
        }
    }

    return o;
}

function restoreFunctions(o)
{
    for (var p in p5.prototype) 
    {
        if (typeof p5.prototype[p] === 'function') 
        {
            window[p] = o[p];
        }
    }
}


function bindP5GlobalFunctions(oGraphics)
{
    for (var p in p5.prototype) 
    {
        if (typeof p5.prototype[p] === 'function') 
        {
            window[p] = p5.prototype[p].bind(oGraphics);
        }
    }
